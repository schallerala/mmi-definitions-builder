\begin{exl}
\label{exl:0n1n}
Let $V = \{S\}$, $T = \{0, 1\}$, $P = \{S \to 0S1, S \to \epsilon\}$.
From
\[
S \Rightarrow 0S1 \Rightarrow 00S11 \Rightarrow \cdots \Rightarrow 0^{n-1}S1^{n-1} \Rightarrow 0^n1^n
\]
we see that $L(G) = \{0^n1^n \mid n \ge 0\}$.
\end{exl}
