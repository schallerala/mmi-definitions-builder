\begin{thm}[Inclusion-exclusion formula]
\label{thm:IncExc}
\begin{multline*}
|A_1 \cup \cdots \cup A_n| = \sum_{i=1}^n |A_i| - \sum_{1 \le i < j \le n} |A_i \cap A_j| 
+ \sum_{1 \le i < j < k \le n} |A_i \cap A_j \cap A_k| - \cdots \\
\cdots +(-1)^{n-1} |A_1 \cap \cdots \cap A_n|
\end{multline*}
\end{thm}
