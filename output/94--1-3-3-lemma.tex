\begin{lem}
\label{lem:BinCoeffRec}
For every $0 \le k \le n$ the following identity holds:
\begin{equation}
\label{eqn:BinCoeffRec}
\binom{n}{k} = \binom{n-1}{k-1} + \binom{n-1}{k}.
\end{equation}
\end{lem}
