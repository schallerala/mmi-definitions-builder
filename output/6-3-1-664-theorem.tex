\begin{thm}
\label{thm:ClosRegLang}
Let $R, R_1, R_2$ be any regular languages in the alphabet $\Sigma$.
Then the languages $R_1 \cup R_2, R_1 \cap R_2, \Sigma^* \setminus R$ are also regular.
\end{thm}
