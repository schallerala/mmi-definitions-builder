\begin{thm}
The sequence of Catalan numbers satisfies the recurrence
\begin{equation}
\label{eqn:CatalanRecursion}
c_{n+1} = \sum_{k=0}^n c_k c_{n-k} = c_0c_n + c_1c_{n-1} + \cdots + c_nc_0,
\end{equation}
where one puts $c_0 = 1$.
\end{thm}
