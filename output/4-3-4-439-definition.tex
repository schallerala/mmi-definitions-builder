\begin{dfn}
A set $X \subset \N^p$ is called \emph{recursively enumerable} if the function
\[
\overline{\mathbf{1}}_X \colon X \to \N, \quad \overline{\mathbf{1}}_X(x) = 1 \text{ for all }x \in X
\]
is recursive.
\end{dfn}
