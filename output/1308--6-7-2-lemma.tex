\begin{lem}
\label{lem:ChomskyTree}
Let $G$ be a grammar in Chomsky form, and $T$ be a parse tree for a word $z \in G$.
If $T$ contains no path of length greater than $i$, then $|z| \le 2^{i-1}$.
\end{lem}
