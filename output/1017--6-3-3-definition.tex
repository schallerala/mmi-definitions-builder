\begin{dfn}
Let $L \subset \Sigma^*$ be a language. The \emph{homomorphic image} of $L$ is the language
\[
h(L) = \{h(w) \mid w \in L\} \subset \Delta^*,
\]
where $h \colon \Sigma^* \to \Delta^*$ is some homomorphism.
\end{dfn}
