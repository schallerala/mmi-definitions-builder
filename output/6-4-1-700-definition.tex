\begin{dfn}
Let $L \subset \Sigma^*$ be a language.
Two words $u, v \in \Sigma^*$ are called \emph{$L$-equivalent} (written as $u \sim_L v$) if
\[
\forall x \in \Sigma^* \text{ either } ux, vx \in L \text{ or } ux, vx \notin L.
\]
\end{dfn}
