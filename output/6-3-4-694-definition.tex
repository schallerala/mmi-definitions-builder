\begin{dfn}
Let $h \colon \Sigma^* \to \Delta^*$ be a homomorphism, and $L \subset \Delta^*$ be a language in the alphabet $\Delta$.
The \emph{inverse homomorphic image} of $L$ is
\[
h^{-1}(L) = \{x \in \Sigma^* \mid h(x) \in L\}.
\]
\end{dfn}
