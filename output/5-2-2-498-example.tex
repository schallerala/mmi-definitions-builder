\begin{exl}
One has
\[
(1 + x)^{-1} = 1 - x + x^2 - \cdots,
\]
\[
(1 - \lambda x)^{-1} = 1 + \lambda x + \lambda^2 x^2 + \cdots,
\]
\[
(1 - x^2)^{-1} = 1 + x^2 + x^4 + \cdots.
\]
\end{exl}
