\begin{dfn}
A theory $T$ is called \emph{recursive} if the set
\[
\sharp T = \{\sharp A \mid A \in T\} \subset \N
\]
is recursive.

A theory $T$ is called \emph{decidable} if the set
\[
\sharp \Th(T) = \{\sharp A \mid A \in \Th(T)\} \subset \N
\]
is recursive.
\end{dfn}
