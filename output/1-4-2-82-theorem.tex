\begin{thm}[Multinomial theorem]
\[
(a_1 + \cdots + a_m)^n = \sum_{\substack{k_1 + \cdots + k_m = n\\ k_1, \ldots, k_m \ge 0}}
\binom{n}{k_1, \ldots, k_m} a_1^{k_1} \cdots a_m^{k_m}
\]
\end{thm}
