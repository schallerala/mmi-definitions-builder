\begin{dfn}
For every $\alpha \in \R$ and every non-negative integer $k$ define the \emph{generalized binomial coefficient} $\binom{\alpha}{k}$ as
\[
\binom{\alpha}{k} = \frac{\alpha(\alpha - 1) \cdots (\alpha - k + 1)}{k!}.
\]
\end{dfn}
