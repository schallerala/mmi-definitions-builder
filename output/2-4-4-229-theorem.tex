\begin{thm}[Hall]
\label{thm:Hall}
Let $G = (X \cup Y, E)$ be a bipartite graph.
Then $G$ contains a matching that covers $X$ if and only if
\begin{equation}
\label{eqn:Hall}
|N(S)| \ge |S| \text{ for all } S \subset X.
\end{equation}
\end{thm}
