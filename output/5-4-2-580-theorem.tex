\begin{thm}
\label{thm:CatalanFormula}
The $n$-th Catalan number is equal to
\begin{equation}
\label{eqn:CatalanFormula}
c_n = \frac1{n+1} \binom{2n}{n}.
\end{equation}
\end{thm}
