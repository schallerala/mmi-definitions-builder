\begin{dfn}
Let $A$ be a proposition and $v$ be a valuation.
If $\hat{v}(A) = 1$, then we say that $v$ \emph{satisfies} $A$ and denote this by $v \vDash A$.
If $\hat{v}(A) = 0$, then we say that $v$ \emph{falsifies} $A$ and denote this by $v \nvDash A$.
\end{dfn}
