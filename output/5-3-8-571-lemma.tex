\begin{lem}
For every $n \ge 1$ we have
\[
q_{n,\mathrm{even}} - q_{n, \mathrm{odd}} =
\begin{cases}
(-1)^k, &\text{if } n = \frac{3k^2 \pm k}2;\\
0, &\text{otherwise.}
\end{cases}
\]
\end{lem}
