\begin{lem}
\label{lem:CatalanGenFunc}
For the generating function $C(x) = c_0 + c_1 x + c_2 x^2 + \cdots$ of the Catalan sequence
the following identity holds:
\[
(C(x))^2 = \frac{C(x) - 1}{x}.
\]
\end{lem}
