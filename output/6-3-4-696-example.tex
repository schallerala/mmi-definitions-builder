\begin{exl}
Let $\Sigma = \{a,b\}$, $\Delta = \{0, 1\}$, and $h(a) = 01$, $h(b) = 10$.
Then
\[
h^{-1}(1001) = \{ba\}, \quad h^{-1}(0011) = \emptyset.
\]
For $L = (00 + 1)^*$ one has $h^{-1}(L) = (ba)^*$ (check this!).
One has $h(h^{-1}(L)) = (1001)^* \ne L$.
\end{exl}
