\begin{thm}
\label{thm:BinomNeg}
For every positive integer $n$ one has
\begin{equation}
\label{eqn:BinomNeg}
(1+x)^{-n} = \sum_{k=0}^\infty \binom{-n}{k} x^k,
\end{equation}
where
\[
\binom{-n}{k} = \frac{-n \cdot (-n-1) \cdot \ldots \cdot (-n-k+1)}{k!}.
\]
\end{thm}
