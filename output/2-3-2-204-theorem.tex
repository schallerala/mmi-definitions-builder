\begin{thm}
\label{thm:3V-6}
Let $G$ be a connected plane graph with $n \ge 3$ vertices.
Then $|E| \le 3|V| - 6$.
Moreover, equality holds if and only if all faces of $G$ are triangles.
\end{thm}
