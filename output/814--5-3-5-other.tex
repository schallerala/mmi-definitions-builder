% \sum_{n=0}^\infty q_nx^n = (1+x^2+x^4+\cdots)(1+x^3+x^6+\cdots)\\
% = \frac{1}{(1-x^2)(1-x^3)\cdots} = (1-x) \sum_{n=0}^\infty p_nx^n
% \end{multline*}

% \begin{multline*}

% \quad\quad - p_0x - p_1x_2 - \cdots\\
