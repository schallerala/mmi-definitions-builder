\begin{thm}[Vandermonde's identity]
\label{thm:Vandermonde}
For every $\alpha, \beta \in \R$ the following holds:
\[
\binom{\alpha + \beta}{k} = \sum_{i=0}^k \binom{\alpha}{i} \binom{\beta}{k-i}.
\]
\end{thm}
