\begin{thm}
\label{thm:BinomSum}
For every $n$ we have
\begin{equation}
\label{eqn:SumBinCoeff}
\sum_{k=1}^n \binom{n}{k} = \binom{n}{0} + \binom{n}{1} + \cdots + \binom{n}{n} = 2^n.
\end{equation}
\end{thm}
