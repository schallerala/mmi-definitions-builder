\begin{thm}
\label{thm:Dollars}
The generating function of the sequence $a_n$ has the following form:
\[
\sum_{n=0}^\infty a_n x^n = \frac{1}{(1-x^9)(1-x^{17})(1-x^{31})}.
\]
\end{thm}
