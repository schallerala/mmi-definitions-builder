\begin{thm}
\label{thm:DualHandshake}
For every plane graph the sum of the degrees of its faces is twice the number of edges:
\[
\sum_{f \in F} \deg f = 2 |E|.
\]
(Here $F$ denotes the set of all faces.)
\end{thm}
